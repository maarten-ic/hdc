103 class descr {
104 public:
105  PYBIND11_NOINLINE descr(const char *text, const std::type_info * const * types) {
106  size_t nChars = len(text), nTypes = len(types);
107  m_text = new char[nChars];
108  m_types = new const std::type_info *[nTypes];
109  memcpy(m_text, text, nChars * sizeof(char));
110  memcpy(m_types, types, nTypes * sizeof(const std::type_info *));
111  }
112 
113  PYBIND11_NOINLINE descr friend operator+(descr &&d1, descr &&d2) {
114  descr r;
115 
116  size_t nChars1 = len(d1.m_text), nTypes1 = len(d1.m_types);
117  size_t nChars2 = len(d2.m_text), nTypes2 = len(d2.m_types);
118 
119  r.m_text = new char[nChars1 + nChars2 - 1];
120  r.m_types = new const std::type_info *[nTypes1 + nTypes2 - 1];
121  memcpy(r.m_text, d1.m_text, (nChars1-1) * sizeof(char));
122  memcpy(r.m_text + nChars1 - 1, d2.m_text, nChars2 * sizeof(char));
123  memcpy(r.m_types, d1.m_types, (nTypes1-1) * sizeof(std::type_info *));
124  memcpy(r.m_types + nTypes1 - 1, d2.m_types, nTypes2 * sizeof(std::type_info *));
125 
126  delete[] d1.m_text; delete[] d1.m_types;
127  delete[] d2.m_text; delete[] d2.m_types;
128 
129  return r;
130  }
131 
132  char *text() { return m_text; }
133  const std::type_info * * types() { return m_types; }

136  PYBIND11_NOINLINE descr() { }
