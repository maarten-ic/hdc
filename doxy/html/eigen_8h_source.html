42  typedef typename Type::Scalar Scalar;
43  static constexpr bool rowMajor = Type::Flags & Eigen::RowMajorBit;

45  bool load(handle src, bool) {
46  array_t<Scalar> buffer(src, true);
47  if (!buffer.check())
48  return false;
49 
50  buffer_info info = buffer.request();
51  if (info.ndim == 1) {
52  typedef Eigen::Stride<Eigen::Dynamic, 0> Strides;
53  if (!Type::IsVectorAtCompileTime &&
54  !(Type::RowsAtCompileTime == Eigen::Dynamic &&
55  Type::ColsAtCompileTime == Eigen::Dynamic))
56  return false;
57 
58  if (Type::SizeAtCompileTime != Eigen::Dynamic &&
59  info.shape[0] != (size_t) Type::SizeAtCompileTime)
60  return false;
61 
62  auto strides = Strides(info.strides[0] / sizeof(Scalar), 0);
63 
64  value = Eigen::Map<Type, 0, Strides>(
65  (Scalar *) info.ptr, info.shape[0], 1, strides);
66  } else if (info.ndim == 2) {
67  typedef Eigen::Stride<Eigen::Dynamic, Eigen::Dynamic> Strides;
68 
69  if ((Type::RowsAtCompileTime != Eigen::Dynamic && info.shape[0] != (size_t) Type::RowsAtCompileTime) ||
70  (Type::ColsAtCompileTime != Eigen::Dynamic && info.shape[1] != (size_t) Type::ColsAtCompileTime))
71  return false;
72 
73  auto strides = Strides(
74  info.strides[rowMajor ? 0 : 1] / sizeof(Scalar),
75  info.strides[rowMajor ? 1 : 0] / sizeof(Scalar));
76 
77  value = Eigen::Map<Type, 0, Strides>(
78  (Scalar *) info.ptr, info.shape[0], info.shape[1], strides);
79  } else {
80  return false;
81  }
82  return true;
83  }
