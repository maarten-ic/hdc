195 struct buffer_info {
196  void *ptr; // Pointer to the underlying storage
197  size_t itemsize; // Size of individual items in bytes
198  size_t size; // Total number of entries
199  std::string format; // For homogeneous buffers, this should be set to format_descriptor<T>::value
200  int ndim; // Number of dimensions
201  std::vector<size_t> shape; // Shape of the tensor (1 entry per dimension)
202  std::vector<size_t> strides; // Number of entries between adjacent entries (for each per dimension)
203 
204  buffer_info() : ptr(nullptr), view(nullptr) {}
205  buffer_info(void *ptr, size_t itemsize, const std::string &format, int ndim,
206  const std::vector<size_t> &shape, const std::vector<size_t> &strides)
207  : ptr(ptr), itemsize(itemsize), size(1), format(format),
208  ndim(ndim), shape(shape), strides(strides) {
209  for (int i=0; i<ndim; ++i) size *= shape[i];
210  }
211 
212  buffer_info(Py_buffer *view)
213  : ptr(view->buf), itemsize(view->itemsize), size(1), format(view->format),
214  ndim(view->ndim), shape(view->ndim), strides(view->ndim), view(view) {
215  for (int i = 0; i < view->ndim; ++i) {
216  shape[i] = (size_t) view->shape[i];
217  strides[i] = (size_t) view->strides[i];
218  size *= shape[i];
219  }
220  }
221 
222  ~buffer_info() {
223  if (view) { PyBuffer_Release(view); delete view; }
224  }
225 private:
226  Py_buffer *view = nullptr;
227 };
