231 template <typename T>
232 struct process_attribute<arg_t<T>> : process_attribute_default<arg_t<T>> {
233  static void init(const arg_t<T> &a, function_record *r) {
234  if (r->class_ && r->args.empty())
235  r->args.emplace_back("self", nullptr, handle());
236 
237  /* Convert keyword value into a Python object */
238  object o = object(detail::type_caster<typename detail::intrinsic_type<T>::type>::cast(
239  a.value, return_value_policy::automatic, handle()), false);
240 
241  if (!o) {
242 #if !defined(NDEBUG)
243  std::string descr(typeid(T).name());
244  detail::clean_type_id(descr);
245  descr = "'" + std::string(a.name) + ": " + descr + "'";
246  if (r->class_) {
247  if (r->name)
248  descr += " in method '" + (std::string) r->class_.str() + "." + (std::string) r->name + "'";
249  else
250  descr += " in method of '" + (std::string) r->class_.str() + "'";
251  } else if (r->name) {
252  descr += " in function named '" + (std::string) r->name + "'";
253  }
254  pybind11_fail("arg(): could not convert default keyword argument "
255  + descr + " into a Python object (type not registered yet?)");
256 #else
257  pybind11_fail("arg(): could not convert default keyword argument "
258  "into a Python object (type not registered yet?). "
259  "Compile in debug mode for more information.");
260 #endif
261  }
262  r->args.emplace_back(a.name, a.descr, o.release());
263  }
264 };
