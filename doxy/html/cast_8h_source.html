33 PYBIND11_NOINLINE inline internals &get_internals() {
34  static internals *internals_ptr = nullptr;
35  if (internals_ptr)
36  return *internals_ptr;
37  handle builtins(PyEval_GetBuiltins());
38  const char *id = PYBIND11_INTERNALS_ID;
39  capsule caps(builtins[id]);
40  if (caps.check()) {
41  internals_ptr = caps;
42  } else {
43  internals_ptr = new internals();

49  internals_ptr->istate = tstate->interp;
50  #endif
51  builtins[id] = capsule(internals_ptr);
52  }
53  return *internals_ptr;
54 }

121 class type_caster_generic {
122 public:
123  PYBIND11_NOINLINE type_caster_generic(const std::type_info &type_info)
124  : typeinfo(get_type_info(type_info)) { }

126  PYBIND11_NOINLINE bool load(handle src, bool convert) {
127  if (!src || !typeinfo)
128  return false;
129  if (src.ptr() == Py_None) {
130  value = nullptr;
131  return true;
132  } else if (PyType_IsSubtype(Py_TYPE(src.ptr()), typeinfo->type)) {
133  value = ((instance<void> *) src.ptr())->value;
134  return true;
135  }
136  if (convert) {
137  for (auto &converter : typeinfo->implicit_conversions) {
138  temp = object(converter(src.ptr(), typeinfo->type), false);
139  if (load(temp, false))
140  return true;
141  }
142  }
143  return false;
144  }

146  PYBIND11_NOINLINE static handle cast(const void *_src, return_value_policy policy, handle parent,
147  const std::type_info *type_info,
148  const std::type_info *type_info_backup,
149  void *(*copy_constructor)(const void *),
150  void *(*move_constructor)(const void *),
151  const void *existing_holder = nullptr) {
152  void *src = const_cast<void *>(_src);
153  if (src == nullptr)
154  return handle(Py_None).inc_ref();
155 
156  // avoid an issue with internal references matching their parent's address
157  bool dont_cache = policy == return_value_policy::reference_internal &&
158  parent && ((instance<void> *) parent.ptr())->value == (void *) src;
159 
160  auto& internals = get_internals();
161  auto it_instance = internals.registered_instances.find(src);
162  if (it_instance != internals.registered_instances.end() && !dont_cache)
163  return handle((PyObject *) it_instance->second).inc_ref();
164 
165  auto it = internals.registered_types_cpp.find(std::type_index(*type_info));
166  if (it == internals.registered_types_cpp.end()) {
167  type_info = type_info_backup;
168  it = internals.registered_types_cpp.find(std::type_index(*type_info));
169  }
170 
171  if (it == internals.registered_types_cpp.end()) {
172  std::string tname = type_info->name();
173  detail::clean_type_id(tname);
174  std::string msg = "Unregistered type : " + tname;
175  PyErr_SetString(PyExc_TypeError, msg.c_str());
176  return handle();
177  }
178 
179  auto tinfo = (const detail::type_info *) it->second;
180  object inst(PyType_GenericAlloc(tinfo->type, 0), false);
181 
182  auto wrapper = (instance<void> *) inst.ptr();
183 
184  wrapper->value = src;
185  wrapper->owned = true;
186  wrapper->parent = nullptr;
187 
188  if (policy == return_value_policy::automatic)
189  policy = return_value_policy::take_ownership;
190  else if (policy == return_value_policy::automatic_reference)
191  policy = return_value_policy::reference;
192 
193  if (policy == return_value_policy::copy) {
194  wrapper->value = copy_constructor(wrapper->value);
195  if (wrapper->value == nullptr)
196  throw cast_error("return_value_policy = copy, but the object is non-copyable!");
197  } else if (policy == return_value_policy::move) {
198  wrapper->value = move_constructor(wrapper->value);
199  if (wrapper->value == nullptr)
200  wrapper->value = copy_constructor(wrapper->value);
201  if (wrapper->value == nullptr)
202  throw cast_error("return_value_policy = move, but the object is neither movable nor copyable!");
203  } else if (policy == return_value_policy::reference) {
204  wrapper->owned = false;
205  } else if (policy == return_value_policy::reference_internal) {
206  wrapper->owned = false;
207  wrapper->parent = parent.inc_ref().ptr();
208  }
209 
210  tinfo->init_holder(inst.ptr(), existing_holder);
211  if (!dont_cache)
212  internals.registered_instances[wrapper->value] = inst.ptr();
213 
214  return inst.release();
215  }
216 
217 protected:
218  const type_info *typeinfo = nullptr;
219  void *value = nullptr;
220  object temp;
221 };

259 protected:
260  typedef void *(*Constructor)(const void *stream);
261 #if !defined(_MSC_VER)
262  /* Only enabled when the types are {copy,move}-constructible *and* when the type
263  does not have a private operator new implementaton. */
264  template <typename T = type> static auto make_copy_constructor(const T *value) -> decltype(new T(*value), Constructor(nullptr)) {
265  return [](const void *arg) -> void * { return new T(*((const T *) arg)); }; }
266  template <typename T = type> static auto make_move_constructor(const T *value) -> decltype(new T(std::move(*((T *) value))), Constructor(nullptr)) {
267  return [](const void *arg) -> void * { return (void *) new T(std::move(*((T *) arg))); }; }
268 #else
269  /* Visual Studio 2015's SFINAE implementation doesn't yet handle the above robustly in all situations.
270  Use a workaround that only tests for constructibility for now. */
271  template <typename T = type, typename = typename std::enable_if<std::is_copy_constructible<T>::value>::type>
272  static Constructor make_copy_constructor(const T *value) {
273  return [](const void *arg) -> void * { return new T(*((const T *)arg)); }; }
274  template <typename T = type, typename = typename std::enable_if<std::is_move_constructible<T>::value>::type>
275  static Constructor make_move_constructor(const T *value) {
276  return [](const void *arg) -> void * { return (void *) new T(std::move(*((T *)arg))); }; }
277 #endif
278  static Constructor make_copy_constructor(...) { return nullptr; }
279  static Constructor make_move_constructor(...) { return nullptr; }

312 template <typename T>
313 struct type_caster<
314  T, typename std::enable_if<std::is_integral<T>::value ||
315  std::is_floating_point<T>::value>::type> {
316  typedef typename std::conditional<sizeof(T) <= sizeof(long), long, long long>::type _py_type_0;
317  typedef typename std::conditional<std::is_signed<T>::value, _py_type_0, typename std::make_unsigned<_py_type_0>::type>::type _py_type_1;
318  typedef typename std::conditional<std::is_floating_point<T>::value, double, _py_type_1>::type py_type;
319 public:
320 
321  bool load(handle src, bool) {
322  py_type py_value;
323 
324  if (!src) {
325  return false;
326  } if (std::is_floating_point<T>::value) {
327  py_value = (py_type) PyFloat_AsDouble(src.ptr());
328  } else if (sizeof(T) <= sizeof(long)) {
329  if (std::is_signed<T>::value)
330  py_value = (py_type) PyLong_AsLong(src.ptr());
331  else
332  py_value = (py_type) PyLong_AsUnsignedLong(src.ptr());
333  } else {
334  if (std::is_signed<T>::value)
335  py_value = (py_type) PYBIND11_LONG_AS_LONGLONG(src.ptr());
336  else
337  py_value = (py_type) PYBIND11_LONG_AS_UNSIGNED_LONGLONG(src.ptr());
338  }
339 
340  if ((py_value == (py_type) -1 && PyErr_Occurred()) ||
341  (std::is_integral<T>::value && sizeof(py_type) != sizeof(T) &&
342  (py_value < (py_type) std::numeric_limits<T>::min() ||
343  py_value > (py_type) std::numeric_limits<T>::max()))) {
344  PyErr_Clear();
345  return false;
346  }
347 
348  value = (T) py_value;
349  return true;
350  }
351 
352  static handle cast(T src, return_value_policy /* policy */, handle /* parent */) {
353  if (std::is_floating_point<T>::value) {
354  return PyFloat_FromDouble((double) src);
355  } else if (sizeof(T) <= sizeof(long)) {
356  if (std::is_signed<T>::value)
357  return PyLong_FromLong((long) src);
358  else
359  return PyLong_FromUnsignedLong((unsigned long) src);
360  } else {
361  if (std::is_signed<T>::value)
362  return PyLong_FromLongLong((long long) src);
363  else
364  return PyLong_FromUnsignedLongLong((unsigned long long) src);
365  }
366  }

369  return cast(*src, policy, parent);
370  }
