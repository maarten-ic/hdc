73  template <typename Func, typename Return, typename... Args, typename... Extra>
74  void initialize(Func &&f, Return (*)(Args...), const Extra&... extra) {
75  struct capture { typename std::remove_reference<Func>::type f; };
76 
77  /* Store the function including any extra state it might have (e.g. a lambda capture object) */
78  auto rec = new detail::function_record();
79 
80  /* Store the capture object directly in the function record if there is enough space */
81  if (sizeof(capture) <= sizeof(rec->data)) {
82  new ((capture *) &rec->data) capture { std::forward<Func>(f) };
83  if (!std::is_trivially_destructible<Func>::value)
84  rec->free_data = [](detail::function_record *r) { ((capture *) &r->data)->~capture(); };
85  } else {
86  rec->data[0] = new capture { std::forward<Func>(f) };
87  rec->free_data = [](detail::function_record *r) { delete ((capture *) r->data[0]); };
88  }
89 
90  /* Type casters for the function arguments and return value */
91  typedef detail::type_caster<typename std::tuple<Args...>> cast_in;
92  typedef detail::type_caster<typename std::conditional<
93  std::is_void<Return>::value, detail::void_type,
94  typename detail::intrinsic_type<Return>::type>::type> cast_out;
95 
96  /* Dispatch code which converts function arguments and performs the actual function call */
97  rec->impl = [](detail::function_record *rec, handle args, handle kwargs, handle parent) -> handle {
98  cast_in args_converter;
99 
100  /* Try to cast the function arguments into the C++ domain */
101  if (!args_converter.load_args(args, kwargs, true))
102  return PYBIND11_TRY_NEXT_OVERLOAD;
103 
104  /* Invoke call policy pre-call hook */
105  detail::process_attributes<Extra...>::precall(args);
106 
107  /* Get a pointer to the capture object */
108  capture *cap = (capture *) (sizeof(capture) <= sizeof(rec->data)
109  ? &rec->data : rec->data[0]);
110 
111  /* Perform the functioncall */
112  handle result = cast_out::cast(args_converter.template call<Return>(cap->f),
113  rec->policy, parent);
114 
115  /* Invoke call policy post-call hook */
116  detail::process_attributes<Extra...>::postcall(args, result);
117 
118  return result;
119  };
120 
121  /* Process any user-provided function attributes */
122  detail::process_attributes<Extra...>::init(extra..., rec);
123 
124  /* Generate a readable signature describing the function's arguments and return value types */
125  using detail::descr;
126  PYBIND11_DESCR signature = cast_in::name() + detail::_(" -> ") + cast_out::name();
127 
128  /* Register the function with Python from generic (non-templated) code */
129  initialize_generic(rec, signature.text(), signature.types(), sizeof...(Args));
130 
131  if (cast_in::has_args) rec->has_args = true;
132  if (cast_in::has_kwargs) rec->has_kwargs = true;
133  }

136  void initialize_generic(detail::function_record *rec, const char *text,
137  const std::type_info *const *types, int args) {
138 
139  /* Create copies of all referenced C-style strings */
140  rec->name = strdup(rec->name ? rec->name : "");
141  if (rec->doc) rec->doc = strdup(rec->doc);
142  for (auto &a: rec->args) {
143  if (a.name)
144  a.name = strdup(a.name);
145  if (a.descr)
146  a.descr = strdup(a.descr);
147  else if (a.value)
148  a.descr = strdup(((std::string) ((object) handle(a.value).attr("__repr__"))().str()).c_str());
149  }
150  auto const &registered_types = detail::get_internals().registered_types_cpp;
151 
152  /* Generate a proper function signature */
153  std::string signature;
154  size_t type_depth = 0, char_index = 0, type_index = 0, arg_index = 0;
155  while (true) {
156  char c = text[char_index++];
157  if (c == '\0')
158  break;
159 
160  if (c == '{') {
161  if (type_depth == 1 && arg_index < rec->args.size()) {
162  signature += rec->args[arg_index].name;
163  signature += " : ";
164  }
165  ++type_depth;
166  } else if (c == '}') {
167  --type_depth;
168  if (type_depth == 1 && arg_index < rec->args.size()) {
169  if (rec->args[arg_index].descr) {
170  signature += " = ";
171  signature += rec->args[arg_index].descr;
172  }
173  arg_index++;
174  }
175  } else if (c == '%') {
176  const std::type_info *t = types[type_index++];
177  if (!t)
178  pybind11_fail("Internal error while parsing type signature (1)");
179  auto it = registered_types.find(std::type_index(*t));
180  if (it != registered_types.end()) {
181  signature += ((const detail::type_info *) it->second)->type->tp_name;
182  } else {
183  std::string tname(t->name());
184  detail::clean_type_id(tname);
185  signature += tname;
186  }
187  } else {
188  signature += c;
189  }
190  }
191  if (type_depth != 0 || types[type_index] != nullptr)
192  pybind11_fail("Internal error while parsing type signature (2)");
193 
194  #if !defined(PYBIND11_CPP14)
195  delete[] types;
196  delete[] text;
197  #endif
198 
199 #if PY_MAJOR_VERSION < 3
200  if (strcmp(rec->name, "__next__") == 0) {
201  std::free(rec->name);
202  rec->name = strdup("next");
203  } else if (strcmp(rec->name, "__bool__") == 0) {
204  std::free(rec->name);
205  rec->name = strdup("__nonzero__");
206  }
207 #endif
208 
209  if (!rec->args.empty() && (int) rec->args.size() != args)
210  pybind11_fail(
211  "cpp_function(): function \"" + std::string(rec->name) + "\" takes " +
212  std::to_string(args) + " arguments, but " + std::to_string(rec->args.size()) +
213  " pybind11::arg entries were specified!");
214 
215  rec->signature = strdup(signature.c_str());
216  rec->args.shrink_to_fit();
217  rec->is_constructor = !strcmp(rec->name, "__init__") || !strcmp(rec->name, "__setstate__");
218  rec->has_args = false;
219  rec->has_kwargs = false;
220  rec->nargs = (uint16_t) args;
221 
222 #if PY_MAJOR_VERSION < 3
223  if (rec->sibling && PyMethod_Check(rec->sibling.ptr()))
224  rec->sibling = PyMethod_GET_FUNCTION(rec->sibling.ptr());
225 #endif
226 
227  detail::function_record *chain = nullptr, *chain_start = rec;
228  if (rec->sibling && PyCFunction_Check(rec->sibling.ptr())) {
229  capsule rec_capsule(PyCFunction_GetSelf(rec->sibling.ptr()), true);
230  chain = (detail::function_record *) rec_capsule;
231  /* Never append a method to an overload chain of a parent class;
232  instead, hide the parent's overloads in this case */
233  if (chain->class_ != rec->class_)
234  chain = nullptr;
235  }
236 
237  if (!chain) {
238  /* No existing overload was found, create a new function object */
239  rec->def = new PyMethodDef();
240  memset(rec->def, 0, sizeof(PyMethodDef));
241  rec->def->ml_name = rec->name;
242  rec->def->ml_meth = reinterpret_cast<PyCFunction>(*dispatcher);
243  rec->def->ml_flags = METH_VARARGS | METH_KEYWORDS;
244 
245  capsule rec_capsule(rec, [](PyObject *o) {
246  destruct((detail::function_record *) PyCapsule_GetPointer(o, nullptr));
247  });
248 
249  object scope_module;
250  if (rec->scope) {
251  scope_module = (object) rec->scope.attr("__module__");
252  if (!scope_module)
253  scope_module = (object) rec->scope.attr("__name__");
254  }
255 
256  m_ptr = PyCFunction_NewEx(rec->def, rec_capsule.ptr(), scope_module.ptr());
257  if (!m_ptr)
258  pybind11_fail("cpp_function::cpp_function(): Could not allocate function object");
259  } else {
260  /* Append at the end of the overload chain */
261  m_ptr = rec->sibling.ptr();
262  inc_ref();
263  chain_start = chain;
264  while (chain->next)
265  chain = chain->next;
266  chain->next = rec;
267  }
268 
269  std::string signatures;
270  int index = 0;
271  /* Create a nice pydoc rec including all signatures and
272  docstrings of the functions in the overload chain */
273  if (chain) {
274  // First a generic signature
275  signatures += rec->name;
276  signatures += "(*args, **kwargs)\n";
277  signatures += "Overloaded function.\n\n";
278  }
279  // Then specific overload signatures
280  for (auto it = chain_start; it != nullptr; it = it->next) {
281  if (chain)
282  signatures += std::to_string(++index) + ". ";
283  signatures += rec->name;
284  signatures += it->signature;
285  signatures += "\n";
286  if (it->doc && strlen(it->doc) > 0) {
287  signatures += "\n";
288  signatures += it->doc;
289  signatures += "\n";
290  }
291  if (it->next)
292  signatures += "\n";
293  }
294 
295  /* Install docstring */
296  PyCFunctionObject *func = (PyCFunctionObject *) m_ptr;
297  if (func->m_ml->ml_doc)
298  std::free((char *) func->m_ml->ml_doc);
299  func->m_ml->ml_doc = strdup(signatures.c_str());
300 
301  if (rec->class_) {
302  m_ptr = PYBIND11_INSTANCE_METHOD_NEW(m_ptr, rec->class_.ptr());
303  if (!m_ptr)
304  pybind11_fail("cpp_function::cpp_function(): Could not allocate instance method object");
305  Py_DECREF(func);
306  }
307  }

310  static void destruct(detail::function_record *rec) {
311  while (rec) {
312  detail::function_record *next = rec->next;
313  if (rec->free_data)
314  rec->free_data(rec);
315  std::free((char *) rec->name);
316  std::free((char *) rec->doc);
317  std::free((char *) rec->signature);
318  for (auto &arg: rec->args) {
319  std::free((char *) arg.name);
320  std::free((char *) arg.descr);
321  arg.value.dec_ref();
322  }
323  if (rec->def) {
324  std::free((char *) rec->def->ml_doc);
325  delete rec->def;
326  }
327  delete rec;
328  rec = next;
329  }

333  static PyObject *dispatcher(PyObject *self, PyObject *args, PyObject *kwargs) {
334  /* Iterator over the list of potentially admissible overloads */
335  detail::function_record *overloads = (detail::function_record *) PyCapsule_GetPointer(self, nullptr),
336  *it = overloads;
337 
338  /* Need to know how many arguments + keyword arguments there are to pick the right overload */
339  size_t nargs = PyTuple_GET_SIZE(args),
340  nkwargs = kwargs ? PyDict_Size(kwargs) : 0;
341 
342  handle parent = nargs > 0 ? PyTuple_GET_ITEM(args, 0) : nullptr,
343  result = PYBIND11_TRY_NEXT_OVERLOAD;
344  try {
345  for (; it != nullptr; it = it->next) {
346  tuple args_(args, true);
347  size_t kwargs_consumed = 0;
348 
349  /* For each overload:
350  1. If the required list of arguments is longer than the
351  actually provided amount, create a copy of the argument
352  list and fill in any available keyword/default arguments.
353  2. Ensure that all keyword arguments were "consumed"
354  3. Call the function call dispatcher (function_record::impl)
355  */
356  size_t nargs_ = nargs;
357  if (nargs < it->args.size()) {
358  nargs_ = it->args.size();
359  args_ = tuple(nargs_);
360  for (size_t i = 0; i < nargs; ++i) {
361  handle item = PyTuple_GET_ITEM(args, i);
362  PyTuple_SET_ITEM(args_.ptr(), i, item.inc_ref().ptr());
363  }
364 
365  int arg_ctr = 0;
366  for (auto const &it2 : it->args) {
367  int index = arg_ctr++;
368  if (PyTuple_GET_ITEM(args_.ptr(), index))
369  continue;
370 
371  handle value;
372  if (kwargs)
373  value = PyDict_GetItemString(kwargs, it2.name);
374 
375  if (value)
376  kwargs_consumed++;
377  else if (it2.value)
378  value = it2.value;
379 
380  if (value) {
381  PyTuple_SET_ITEM(args_.ptr(), index, value.inc_ref().ptr());
382  } else {
383  kwargs_consumed = (size_t) -1; /* definite failure */
384  break;
385  }
386  }
387  }
388 
389  try {
390  if ((kwargs_consumed == nkwargs || it->has_kwargs) &&
391  (nargs_ == it->nargs || it->has_args))
392  result = it->impl(it, args_, kwargs, parent);
393  } catch (cast_error &) {
394  result = PYBIND11_TRY_NEXT_OVERLOAD;
395  }
396 
397  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
398  break;
399  }
400  } catch (const error_already_set &) { return nullptr;
401  } catch (const index_error &e) { PyErr_SetString(PyExc_IndexError, e.what()); return nullptr;
402  } catch (const value_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return nullptr;
403  } catch (const stop_iteration &e) { PyErr_SetString(PyExc_StopIteration, e.what()); return nullptr;
404  } catch (const std::bad_alloc &e) { PyErr_SetString(PyExc_MemoryError, e.what()); return nullptr;
405  } catch (const std::domain_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return nullptr;
406  } catch (const std::invalid_argument &e) { PyErr_SetString(PyExc_ValueError, e.what()); return nullptr;
407  } catch (const std::length_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return nullptr;
408  } catch (const std::out_of_range &e) { PyErr_SetString(PyExc_IndexError, e.what()); return nullptr;
409  } catch (const std::range_error &e) { PyErr_SetString(PyExc_ValueError, e.what()); return nullptr;
410  } catch (const std::exception &e) { PyErr_SetString(PyExc_RuntimeError, e.what()); return nullptr;
411  } catch (...) {
412  PyErr_SetString(PyExc_RuntimeError, "Caught an unknown exception!");
413  return nullptr;
414  }
415 
416  if (result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
417  std::string msg = "Incompatible function arguments. The "
418  "following argument types are supported:\n";
419  int ctr = 0;
420  for (detail::function_record *it2 = overloads; it2 != nullptr; it2 = it2->next) {
421  msg += " "+ std::to_string(++ctr) + ". ";
422  msg += it2->signature;
423  msg += "\n";
424  }
425  PyErr_SetString(PyExc_TypeError, msg.c_str());
426  return nullptr;
427  } else if (!result) {
428  std::string msg = "Unable to convert function return value to a "
429  "Python type! The signature was\n\t";
430  msg += it->signature;
431  PyErr_SetString(PyExc_TypeError, msg.c_str());
432  return nullptr;
433  } else {
434  if (overloads->is_constructor) {
435  /* When a constructor ran successfully, the corresponding
436  holder type (e.g. std::unique_ptr) must still be initialized. */
437  PyObject *inst = PyTuple_GET_ITEM(args, 0);
438  auto tinfo = detail::get_type_info(Py_TYPE(inst));
439  tinfo->init_holder(inst, nullptr);
440  }
441  return result.ptr();
442  }
443  }

499 public:
500  PYBIND11_OBJECT_DEFAULT(generic_type, object, PyType_Check)
501 protected:
502  void initialize(type_record *rec) {
503  if (rec->base_type) {
504  if (rec->base_handle)
505  pybind11_fail("generic_type: specified base type multiple times!");
506  rec->base_handle = detail::get_type_handle(*(rec->base_type));
507  if (!rec->base_handle) {
508  std::string tname(rec->base_type->name());
509  detail::clean_type_id(tname);
510  pybind11_fail("generic_type: type \"" + std::string(rec->name) +
511  "\" referenced unknown base type \"" + tname + "\"");
512  }
513  }
514 
515  object type_holder(PyType_Type.tp_alloc(&PyType_Type, 0), false);
516  object name(PYBIND11_FROM_STRING(rec->name), false);
517  auto type = (PyHeapTypeObject*) type_holder.ptr();
518 
519  if (!type_holder || !name)
520  pybind11_fail("generic_type: unable to create type object!");
521 
522  /* Register supplemental type information in C++ dict */
523  auto &internals = get_internals();
524  detail::type_info *tinfo = new detail::type_info();
525  tinfo->type = (PyTypeObject *) type;
526  tinfo->type_size = rec->type_size;
527  tinfo->init_holder = rec->init_holder;
528  internals.registered_types_cpp[std::type_index(*(rec->type))] = tinfo;
529  internals.registered_types_py[type] = tinfo;
530 
531  object scope_module;
532  if (rec->scope) {
533  scope_module = (object) rec->scope.attr("__module__");
534  if (!scope_module)
535  scope_module = (object) rec->scope.attr("__name__");
536  }
537 
538  std::string full_name = (scope_module ? ((std::string) scope_module.str() + "." + rec->name)
539  : std::string(rec->name));
540  /* Basic type attributes */
541  type->ht_type.tp_name = strdup(full_name.c_str());
542  type->ht_type.tp_basicsize = rec->instance_size;
543  type->ht_type.tp_base = (PyTypeObject *) rec->base_handle.ptr();
544  rec->base_handle.inc_ref();
545 
546 #if PY_MAJOR_VERSION >= 3 && PY_MINOR_VERSION >= 3
547  /* Qualified names for Python >= 3.3 */
548  object scope_qualname;
549  if (rec->scope)
550  scope_qualname = (object) rec->scope.attr("__qualname__");
551  if (scope_qualname) {
552  type->ht_qualname = PyUnicode_FromFormat(
553  "%U.%U", scope_qualname.ptr(), name.ptr());
554  } else {
555  type->ht_qualname = name.ptr();
556  name.inc_ref();
557  }
558 #endif
559  type->ht_name = name.release().ptr();
560 
561  /* Supported protocols */
562  type->ht_type.tp_as_number = &type->as_number;
563  type->ht_type.tp_as_sequence = &type->as_sequence;
564  type->ht_type.tp_as_mapping = &type->as_mapping;
565 
566  /* Supported elementary operations */
567  type->ht_type.tp_init = (initproc) init;
568  type->ht_type.tp_new = (newfunc) new_instance;
569  type->ht_type.tp_dealloc = rec->dealloc;
570 
571  /* Support weak references (needed for the keep_alive feature) */
572  type->ht_type.tp_weaklistoffset = offsetof(instance_essentials<void>, weakrefs);
573 
574  /* Flags */
575  type->ht_type.tp_flags |= Py_TPFLAGS_DEFAULT | Py_TPFLAGS_BASETYPE | Py_TPFLAGS_HEAPTYPE;
576 #if PY_MAJOR_VERSION < 3
577  type->ht_type.tp_flags |= Py_TPFLAGS_CHECKTYPES;
578 #endif
579  type->ht_type.tp_flags &= ~Py_TPFLAGS_HAVE_GC;
580 
581  if (rec->doc) {
582  /* Allocate memory for docstring (using PyObject_MALLOC, since
583  Python will free this later on) */
584  size_t size = strlen(rec->doc) + 1;
585  type->ht_type.tp_doc = (char *) PyObject_MALLOC(size);
586  memcpy((void *) type->ht_type.tp_doc, rec->doc, size);

960 PYBIND11_NOINLINE inline void keep_alive_impl(int Nurse, int Patient, handle args, handle ret) {
961  /* Clever approach based on weak references taken from Boost.Python */
962  handle nurse (Nurse > 0 ? PyTuple_GetItem(args.ptr(), Nurse - 1) : ret.ptr());
963  handle patient(Patient > 0 ? PyTuple_GetItem(args.ptr(), Patient - 1) : ret.ptr());
964 
965  if (!nurse || !patient)
966  pybind11_fail("Could not activate keep_alive!");
967 
968  if (patient.ptr() == Py_None)
969  return; /* Nothing to keep alive */
970

972  [patient](handle weakref) { patient.dec_ref(); weakref.dec_ref(); });
973

975 
976  patient.inc_ref(); /* reference patient and leak the weak reference */

1026 /* The functions below essentially reproduce the PyGILState_* API using a RAII
1027  * pattern, but there are a few important differences:
1028  *
1029  * 1. When acquiring the GIL from an non-main thread during the finalization
1030  * phase, the GILState API blindly terminates the calling thread, which
1031  * is often not what is wanted. This API does not do this.
1032  *
1033  * 2. The gil_scoped_release function can optionally cut the relationship
1034  * of a PyThreadState and its associated thread, which allows moving it to
1035  * another thread (this is a fairly rare/advanced use case).
1036  *
1037  * 3. The reference count of an acquired thread state can be controlled. This
1038  * can be handy to prevent cases where callbacks issued from an external
1039  * thread would otherwise constantly construct and destroy thread state data
1040  * structures.
1041  */
1042 
1043 class gil_scoped_acquire {
1044 public:
1045  PYBIND11_NOINLINE gil_scoped_acquire() {
1046  auto const &internals = detail::get_internals();
1047  tstate = (PyThreadState *) PyThread_get_key_value(internals.tstate);
1048 
1049  if (!tstate) {
1050  tstate = PyThreadState_New(internals.istate);
1051  #if !defined(NDEBUG)
1052  if (!tstate)
1053  pybind11_fail("scoped_acquire: could not create thread state!");
1054  #endif
1055  tstate->gilstate_counter = 0;
1056  #if PY_MAJOR_VERSION < 3
1057  PyThread_delete_key_value(internals.tstate);
1058  #endif
1059  PyThread_set_key_value(internals.tstate, tstate);
1060  } else {
1061  release = detail::get_thread_state_unchecked() != tstate;
1062  }
1063 
1064  if (release) {
1065  /* Work around an annoying assertion in PyThreadState_Swap */
1066  #if defined(Py_DEBUG)
1067  PyInterpreterState *interp = tstate->interp;
1068  tstate->interp = nullptr;
1069  #endif
1070  PyEval_AcquireThread(tstate);
1071  #if defined(Py_DEBUG)
1072  tstate->interp = interp;
1073  #endif
1074  }
1075 
1076  inc_ref();
1077  }
1078 
1079  void inc_ref() {
1080  ++tstate->gilstate_counter;
1081  }
1082 
1083  PYBIND11_NOINLINE void dec_ref() {
1084  --tstate->gilstate_counter;
1085  #if !defined(NDEBUG)
1086  if (detail::get_thread_state_unchecked() != tstate)
1087  pybind11_fail("scoped_acquire::dec_ref(): thread state must be current!");
1088  if (tstate->gilstate_counter < 0)
1089  pybind11_fail("scoped_acquire::dec_ref(): reference count underflow!");
1090  #endif
1091  if (tstate->gilstate_counter == 0) {
1092  #if !defined(NDEBUG)
1093  if (!release)
1094  pybind11_fail("scoped_acquire::dec_ref(): internal error!");
1095  #endif
1096  PyThreadState_Clear(tstate);
1097  PyThreadState_DeleteCurrent();
1098  PyThread_delete_key_value(detail::get_internals().tstate);
1099  release = false;
1100  }
1101  }
1102 
1103  PYBIND11_NOINLINE ~gil_scoped_acquire() {
1104  dec_ref();
1105  if (release)
1106  PyEval_SaveThread();
1107  }
1108 private:
1109  PyThreadState *tstate = nullptr;
1110  bool release = true;
